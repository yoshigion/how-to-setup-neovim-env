[Supported Newer GLIBC]
(install neovim stable from ppa)
sudo add-apt-repository ppa:neovim-ppa/stable
sudo apt install neovim

(install recent stable neovim at /opt/neovim/)
sudo mkdir -p /opt/neovim
sudo chown obayashi.obayashi /opt/neovim
cd /opt/neovim
wget https://github.com/neovim/neovim/releases/download/v0.11.5/nvim-linux-x86_64.tar.gz
tar xvf nvim-linux-x86_64.tar.gz
((copy and edit ex/rview/rvim/view/vimdiff from stable ppa))
cd nvim-linux-x86_64
cp -rp /usr/libexec/neovim ./libexec
cd libexec
find . -name "[a-z]*" -exec sed -i -e "/nvim/s|/usr/bin|/opt/neovim/nvim-linux-x86_64/bin|" {} \;

(setup ~/.bashrc)
vi ~/.bashrc
### for neovim recent
NEOVIM_PATH="/opt/neovim/nvim-linux-x86_64"
export PATH="$NEOVIM_PATH/bin:$PATH"

(setup alternatives)
sudo update-alternatives --install /usr/bin/editor  editor  /opt/neovim/nvim-linux-x86_64/bin/nvim 30
sudo update-alternatives --install /usr/bin/vi      vi      /opt/neovim/nvim-linux-x86_64/bin/nvim 30
sudo update-alternatives --install /usr/bin/vim     vim     /opt/neovim/nvim-linux-x86_64/bin/nvim 30
sudo update-alternatives --install /usr/bin/view    view    /opt/neovim/nvim-linux-x86_64/libexec/view 30
sudo update-alternatives --install /usr/bin/vimdiff vimdiff /opt/neovim/nvim-linux-x86_64/libexec/vimdiff 30

sudo update-alternatives --config editor
sudo update-alternatives --config vi
sudo update-alternatives --config vim
sudo update-alternatives --config view
sudo update-alternatives --config vimdiff
